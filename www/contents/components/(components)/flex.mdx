```tsx preview
<Flex gap="md">
  <For each={["info", "success", "warning", "error"]}>
    {(bg, index) => (
      <Box key={index} bg={bg} color="white" p="md" rounded="l2">
        Box
      </Box>
    )}
  </For>
</Flex>
```

## Usage

:::code-group

```tsx [package]
import { Flex } from "@yamada-ui/react"
```

```tsx [alias]
import { Flex } from "@/components/ui"
```

```tsx [monorepo]
import { Flex } from "@workspace/ui"
```

:::

```tsx
<Flex />
```

### Direction

```tsx preview
<Flex direction="column" gap="md">
  <For each={["info", "success", "warning", "error"]}>
    {(bg, index) => (
      <Box key={index} bg={bg} color="white" p="md" rounded="l2">
        Box
      </Box>
    )}
  </For>
</Flex>
```

### Align

```tsx preview
<Flex align="center" gap="md">
  <For each={["sm", "md", "lg"]}>
    {(p, index) => (
      <Box key={index} bg={"info"} color="white" p={p} rounded="l2">
        Box
      </Box>
    )}
  </For>
</Flex>
```

