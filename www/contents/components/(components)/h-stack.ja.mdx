```tsx preview
<HStack>
  <For each={["info", "success", "warning", "error"]}>
    {(bg, index) => (
      <Box key={index} p="md" rounded="4" bg={bg} color="white">
        Box
      </Box>
    )}
  </For>
</HStack>
```

## 使い方

:::code-group

```tsx [package]
import { HStack } from "@yamada-ui/react"
```

```tsx [alias]
import { HStack } from "@/components/ui"
```

```tsx [monorepo]
import { HStack } from "@workspace/ui"
```

:::

```tsx
<HStack />
```

### 要素間のスペースを確保する

```tsx preview
<HStack>
  <Box p="md" rounded="4" bg="info" color="white">
    Box
  </Box>

  <Spacer />

  <Box p="md" rounded="4" bg="error" color="white">
    Box
  </Box>
</HStack>
```
