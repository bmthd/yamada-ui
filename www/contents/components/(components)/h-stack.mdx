```tsx preview
<HStack>
  <For each={["info", "success", "warning", "error"]}>
    {(bg, index) => (
      <Box key={index} p="md" rounded="l2" bg={bg} color="white">
        Box
      </Box>
    )}
  </For>
</HStack>
```

## Usage

:::code-group

```tsx [package]
import { HStack } from "@yamada-ui/react"
```

```tsx [alias]
import { HStack } from "@/components/ui"
```

```tsx [monorepo]
import { HStack } from "@workspace/ui"
```

:::

```tsx
<HStack />
```

### Use spaces

```tsx preview
<HStack>
  <Box p="md" rounded="l2" bg="info" color="white">
    Box
  </Box>

  <Spacer />

  <Box p="md" rounded="l2" bg="error" color="white">
    Box
  </Box>
</HStack>
```
