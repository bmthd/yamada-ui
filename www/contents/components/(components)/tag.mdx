## Usage

:::code-group

```tsx [package]
import { Tag } from "@yamada-ui/react"
```

```tsx [alias]
import { Tag } from "@/components/ui"
```

```tsx [monorepo]
import { Tag } from "@workspaces/ui"
```

:::

```tsx
<Tag />
```

### Change Variants

```tsx preview
<HStack>
  <For each={["solid", "subtle", "surface", "outline"]}>
    {(variant) => (
      <Tag key={variant} variant={variant} colorScheme="primary">
        {variant}
      </Tag>
    )}
  </For>
</HStack>
```

### Change Size

```tsx preview
<HStack>
  <For each={["sm", "md", "lg"]}>
    {(size) => (
      <Tag key={size} size={size} colorScheme="primary">
        {size}
      </Tag>
    )}
  </For>
</HStack>
```

### Change Color Scheme

```tsx preview
<HStack>
  <For each={["success", "warning", "error"]}>
    {(colorScheme) => (
      <Tag key={colorScheme} colorScheme={colorScheme}>
        {colorScheme}
      </Tag>
    )}
  </For>
</HStack>
```

### Use with Icon

```tsx preview
<HStack>
  <Tag startIcon={<PlusIcon />}>startIcon</Tag>
  <Tag endIcon={<PlusIcon />}>endIcon</Tag>
</HStack>
```

### Use Close Button

```tsx preview functional client
const onClose = () => {
  // Handle close action
  console.log("Tag closed")
}

return <Tag onClose={onClose}>Tag</Tag>
```

### Disable Close Button

```tsx preview
<Tag disabled onClose={() => {}}>
  Tag
</Tag>
```
