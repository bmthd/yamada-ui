```tsx preview
<VStack>
  <For each={["info", "success", "warning", "error"]}>
    {(bg, index) => (
      <Box key={index} p="md" rounded="l2" bg={bg} color="white">
        Box
      </Box>
    )}
  </For>
</VStack>
```

## Usage

:::code-group

```tsx [package]
import { VStack } from "@yamada-ui/react"
```

```tsx [alias]
import { VStack } from "@/components/ui"
```

```tsx [monorepo]
import { VStack } from "@workspace/ui"
```

:::

```tsx
<VStack />
```

### Adding Separators Between Child Elements

Use the `separator` prop to add a separator between the child elements.

```tsx preview
<VStack separator={<Separator />}>
  <For each={["info", "success", "warning", "error"]}>
    {(bg, index) => (
      <Box key={index} p="md" rounded="l2" bg={bg} color="white">
        Box
      </Box>
    )}
  </For>
</VStack>
```
